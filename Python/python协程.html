    missionOne 2023/2/12 Python

    本文介绍了什么是python协程，以及如何使用协程并获取返回值

    # 示例代码

    这里假设我们需要向多个API发起请求

    import time
import json
import asyncio
import aiohttp
from concurrent.futures import TimeoutError as CFTimeoutError
from asyncio.exceptions import TimeoutError as AETimeoutError


class Response:
    """模拟requests返回，实现返回类型统一
    """
    def __init__(self, *args, **kwargs) -> None:
        self.text = kwargs.get("text", "")
        self.status_code = kwargs.get("status_code", "")
        self.url = kwargs.get("url", "")


async def send_requests(method, *args, **kwargs):
    url = kwargs.get("url", "")

    # 如果没设置超时时间则给一个默认值
    if "timeout" not in kwargs:
        kwargs["timeout"] = 10

    # 通过data、params作为request参数的情况
    body = kwargs.get("json")
    if body is None:
        body = kwargs.get("params")
    if body is None:
        body = kwargs.get("data", "")

    if isinstance(body, dict):
        kwargs["data"] = json.dumps(body)

    msg = f"start send {method} request to {url}"
    print(msg)
    
    # 发起请求
    start_time = time.perf_counter()
    try:
        async with aiohttp.ClientSession() as session:
            async with getattr(session, str(method).lower())(*args, **kwargs) as resp:
                end_time = time.perf_counter()
                use_time = int((end_time - start_time) * 1000)
                resp_data = await resp.text()
                status_code = resp.status
                msg = f"end send {method} request to {url} got {status_code}, used: {use_time} ms, " \
                      f"response {str(resp_data)[0:250]}"
                print(msg)
            return Response(text=resp_data, status_code=status_code, url=url)
    except (CFTimeoutError, AETimeoutError) as e:
        end_time = time.perf_counter()
        use_time = int((end_time - start_time) * 1000)
        err_msg = f"request to {url} timeout, used {use_time}: {e}"
        print(err_msg)
        return Response(text=err_msg, status_code=500, url=url)
    except Exception as e:  # pylint: disable=broad-except
        end_time = time.perf_counter()
        use_time = int((end_time - start_time) * 1000)
        err_msg = f"request to {url} failed: {e}, used: {use_time}."
        print(err_msg)
        return Response(text=err_msg, status_code=500, url=url)


async def do_things(api_list):
    # 协程并发请求
    tasks = []

    req_headers = {"Content-Type": "application/json"}
    for api in api_list:
        tasks.append(asyncio.ensure_future(
            send_requests(
                "GET", url=api, headers=req_headers, timeout=10
            )
        ))
    for task in asyncio.as_completed(tasks):
        resp = await task
        print(resp.status_code)
        print(resp.text)

    return True


if __name__ == "__main__":
    api_list = [
        "http://api1.test.com/v1/api"
        "http://api2.test.com/v1/api",
        "http://api3.test.com/v1/api",
    ]
    
    # 生成事件循环
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)

    # 生成任务列表并放入事件循环
    result = loop.run_until_complete(do_things(api_list))
